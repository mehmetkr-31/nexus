Server-side utilities live in `@nexus-framework/react/server`. **Do not import this in Client Components.**

```ts
import { createServerNexusClient, getLedgerData, withLedgerAction }
  from '@nexus-framework/react/server';
```

## createServerNexusClient

Reads the session cookie from a `Request` and returns a `NexusClient` scoped to that user.

```ts
// app/contracts/page.tsx (Server Component)
import { cookies } from 'next/headers';
import { createServerNexusClient } from '@nexus-framework/react/server';

const client = await createServerNexusClient({
  request: new Request('http://localhost', {
    headers: { cookie: (await cookies()).toString() },
  }),
  config: {
    ledgerApiUrl: process.env.CANTON_API_URL!,
  },
});
```

## SSR Prefetch + HydrationBoundary

Pre-populate the TanStack Query cache on the server — client components render instantly with no loading state:

```tsx
// app/contracts/page.tsx
import { contractQueryOptions } from '@nexus-framework/react';
import { dehydrate, HydrationBoundary, QueryClient } from '@tanstack/react-query';

export default async function ContractsPage() {
  const client = await createServerNexusClient({ ... });
  const queryClient = new QueryClient();

  await queryClient.prefetchQuery(
    contractQueryOptions({ client, templateId: 'my-package:Iou:Iou' })
  );

  return (
    <HydrationBoundary state={dehydrate(queryClient)}>
      <IouList />  {/* nexus.useContracts() hits cache — zero loading state */}
    </HydrationBoundary>
  );
}
```

## getLedgerData

Shorthand for fetching contracts in a Server Component:

```ts
const { contracts } = await getLedgerData({
  client,
  templateId: 'my-package:Iou:Iou',
  parties: ['Alice::abc'],
});
```

## withLedgerAction

Wraps any ledger operation in a Server Action with typed error handling:

```ts
// app/contracts/actions.ts
'use server';

import { withLedgerAction, createServerNexusClient } from '@nexus-framework/react/server';
import { cookies } from 'next/headers';
import { revalidatePath } from 'next/cache';

export async function createIouAction(formData: FormData) {
  const client = await createServerNexusClient({
    request: new Request('http://localhost', {
      headers: { cookie: (await cookies()).toString() },
    }),
    config: { ledgerApiUrl: process.env.CANTON_API_URL! },
  });

  const result = await withLedgerAction(client, (c) =>
    c.ledger.commands.createContract(
      'my-package:Iou:Iou',
      {
        owner: formData.get('owner') as string,
        amount: formData.get('amount') as string,
        currency: formData.get('currency') as string,
      },
      [formData.get('owner') as string],
    ),
  );

  if (result.success) revalidatePath('/contracts');
  return result;
}
```

### LedgerActionResult

```ts
type LedgerActionResult<T> =
  | { success: true; data: T }
  | { success: false; error: string; code: 'LEDGER_ERROR' | 'UNKNOWN_ERROR' };
```

### Client Component consuming the action

```tsx
"use client";

import { useTransition } from 'react';
import { createIouAction } from '../actions';

export function CreateIouForm() {
  const [isPending, startTransition] = useTransition();

  return (
    <form action={(fd) => startTransition(async () => {
      const result = await createIouAction(fd);
      if (!result.success) alert(result.error);
    })}>
      <input name="owner" required />
      <input name="amount" type="number" required />
      <input name="currency" maxLength={3} required />
      <button type="submit" disabled={isPending}>Create</button>
    </form>
  );
}
```
