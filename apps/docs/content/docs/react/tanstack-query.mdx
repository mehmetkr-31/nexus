`tanstackQueryPlugin()` is a React plugin that adds TanStack Query-powered hooks directly to your `nexus` client instance. Hooks close over the client via closures — no `useContext` calls, no React Context for the Nexus client itself.

## Installation

<Steps>
  <Step>

  ### Install TanStack Query

  ```bash
  bun add @tanstack/react-query
  ```

  </Step>
  <Step>

  ### Add the plugin to createNexusClient

  ```ts title="lib/nexus-client.ts"
  "use client";

  import {
    createNexusClient,
    sandboxAuth,
    tanstackQueryPlugin, // [!code highlight]
  } from '@nexus-framework/react';

  export const nexus = createNexusClient({
    baseUrl: 'http://localhost:7575',
    plugins: [
      sandboxAuth({ secret: 'secret', userId: 'alice', partyId: 'Alice::...' }),
      tanstackQueryPlugin(), // [!code highlight]
    ],
  });
  ```

  After adding the plugin, `nexus.useContracts`, `nexus.useCreateContract`, etc. are all available.

  </Step>
  <Step>

  ### Wrap your app with NexusProvider

  `nexus.NexusProvider` provides the `QueryClientProvider` needed by TanStack Query.

  ```tsx title="app/layout.tsx"
  import { nexus } from '@/lib/nexus-client';

  export default function RootLayout({ children }: { children: React.ReactNode }) {
    return (
      <html>
        <body>
          <nexus.NexusProvider>
            {children}
          </nexus.NexusProvider>
        </body>
      </html>
    );
  }
  ```

  </Step>
</Steps>

## Query Hooks

### useContracts

Fetches active contracts for a Daml template. Data is cached in TanStack Query and revalidated in the background.

```tsx title="components/IouList.tsx"
"use client";

import { nexus } from '@/lib/nexus-client';

interface IouPayload {
  owner: string;
  amount: string;
  currency: string;
}

export function IouList() {
  const { data, isLoading, error, isFetching } = nexus.useContracts<IouPayload>({
    templateId: 'my-package:Iou:Iou',
    parties: ['Alice::122059a10c...'], // optional — defaults to auth party
    enabled: true,                     // optional — set false to pause
    staleTime: 10_000,                 // optional — ms before refetch
  });

  if (isLoading) return <p>Loading…</p>;
  if (error) return <p>Error: {error.message}</p>;

  return (
    <div>
      {isFetching && <span>Refreshing…</span>}
      <ul>
        {data?.contracts.map((c) => (
          <li key={c.contractId}>
            {c.payload.amount} {c.payload.currency}
          </li>
        ))}
      </ul>
    </div>
  );
}
```

### useContractsSuspense

Suspense variant of `useContracts`. Never returns `isLoading` — instead throws a Promise for React's `<Suspense>` boundary. The `data` field is always defined inside the component.

```tsx title="components/IouListSuspense.tsx"
"use client";

import { Suspense } from 'react';
import { nexus } from '@/lib/nexus-client';

function IouListContent() {
  // data is always defined — Suspense handles the loading state
  const { data } = nexus.useContractsSuspense<IouPayload>({
    templateId: 'my-package:Iou:Iou',
  });

  return (
    <ul>
      {data.contracts.map((c) => (
        <li key={c.contractId}>
          {c.payload.amount} {c.payload.currency}
        </li>
      ))}
    </ul>
  );
}

export function IouList() {
  return (
    <Suspense fallback={<p>Loading contracts…</p>}>
      <IouListContent />
    </Suspense>
  );
}
```

### useInterface

Queries contracts through a Daml **interface**, returning the decoded `interfaceView` alongside the underlying contract. This lets you query across multiple template types that implement the same interface.

```tsx title="components/AssetList.tsx"
"use client";

import { nexus } from '@/lib/nexus-client';

interface IAssetView {
  assetName: string;
  totalValue: number;
  issuer: string;
}

export function AssetList() {
  const { data, isLoading } = nexus.useInterface<IAssetView>({
    interfaceId: 'my-package:Asset:IAsset',
    parties: ['Alice::122059a10c...'],
  });

  if (isLoading) return <p>Loading assets…</p>;

  return (
    <ul>
      {data?.contracts.map((c) => (
        <li key={c.contractId}>
          <strong>{c.interfaceView.assetName}</strong>
          {' '}— {c.interfaceView.totalValue} (from {c.templateId.entityName})
        </li>
      ))}
    </ul>
  );
}
```

### usePartyId

Resolves a Canton `userId` to a `partyId`. Result is cached for 5 minutes (matching the `PartyIdResolver` TTL).

```tsx
const { data: partyId, isLoading } = nexus.usePartyId('alice');
// data: "Alice::122059a10c..." | undefined
```

---

### useLedgerEnd

Returns the current ledger offset. Useful for polling-based sync or as a dependency for time-based logic. Refreshes every 2 seconds.

```tsx
const { data: ledgerEnd } = nexus.useLedgerEnd();
console.log('Current offset:', ledgerEnd?.offset);
```

---

### useSynchronizers

Lists the Canton synchronizers connected to this participant. Refreshes every 30 seconds.

```tsx
const { data: synchronizers } = nexus.useSynchronizers();
// data: SynchronizerInfo[] | undefined
```

## Mutation Hooks

### useCreateContract

Creates a Daml contract and automatically invalidates the TanStack Query cache for the affected template, triggering a background refetch.

```tsx title="components/CreateIouForm.tsx"
"use client";

import { useState, useTransition } from 'react';
import { nexus } from '@/lib/nexus-client';

const PARTY_ID = 'Alice::122059a10c...';

export function CreateIouForm() {
  const [amount, setAmount] = useState('100');

  const { mutate: createIou, isPending, error, isSuccess } = nexus.useCreateContract({
    // Automatically invalidate cache for these templates after success
    invalidateTemplates: ['my-package:Iou:Iou'],
    onSuccess: async (result) => {
      console.log('Created in tx:', result.transactionId);
    },
    onError: (err) => {
      console.error('Create failed:', err.message);
    },
  });

  return (
    <form
      onSubmit={(e) => {
        e.preventDefault();
        createIou({
          templateId: 'my-package:Iou:Iou',
          createArguments: { owner: PARTY_ID, amount, currency: 'USD' },
          actAs: [PARTY_ID],
        });
      }}
    >
      <input
        type="number"
        value={amount}
        onChange={(e) => setAmount(e.target.value)}
        min="1"
        required
      />
      {error && <p style={{ color: 'red' }}>{error.message}</p>}
      {isSuccess && <p style={{ color: 'green' }}>Contract created!</p>}
      <button type="submit" disabled={isPending}>
        {isPending ? 'Creating…' : 'Create'}
      </button>
    </form>
  );
}
```

### useExerciseChoice

Exercises a Daml choice and invalidates the cache. Returns `SubmitResult` (no Daml return value). Use `useExerciseAndGetResult` if you need the typed return value.

```tsx
const { mutate: archive, isPending } = nexus.useExerciseChoice({
  invalidateTemplates: ['my-package:Iou:Iou'],
});

// In a button handler:
archive({
  templateId: 'my-package:Iou:Iou',
  contractId: contract.contractId,
  choice: 'Archive',
  choiceArgument: {},
  actAs: ['Alice::122059a10c...'],
});
```

### useExerciseAndGetResult

Exercises a choice and returns the **typed Daml return value** from the exercised event. This is the TanStack Query wrapper around `exerciseAndGetResult` — it also invalidates the cache after success.

```tsx title="components/TransferButton.tsx"
"use client";

import { nexus } from '@/lib/nexus-client';

interface TransferArgs {
  newOwner: string;
}

interface TransferResult {
  newContractId: string;
}

export function TransferButton({ contractId }: { contractId: string }) {
  const { mutate: transfer, isPending, data } = nexus.useExerciseAndGetResult<
    TransferArgs,
    TransferResult
  >({
    invalidateTemplates: ['my-package:Iou:Iou'],
    onSuccess: (result) => {
      console.log('Transferred. New contract:', result.result.newContractId);
    },
  });

  return (
    <div>
      <button
        disabled={isPending}
        onClick={() =>
          transfer({
            templateId: 'my-package:Iou:Iou',
            contractId,
            choice: 'Transfer',
            choiceArgument: { newOwner: 'Bob::xyz...' },
            actAs: ['Alice::122059a10c...'],
          })
        }
      >
        {isPending ? 'Transferring…' : 'Transfer to Bob'}
      </button>
      {data && (
        <p>New contract: {data.result.newContractId.slice(0, 16)}…</p>
      )}
    </div>
  );
}
```

### useLedgerMutation

Generic escape hatch for custom ledger operations. Receives the full `NexusClient` instance so you can compose multiple commands, fetch data mid-mutation, or do anything the core API supports.

```tsx
const { mutate } = nexus.useLedgerMutation({
  mutationFn: async (client, vars: { contractId: string; splitAmount: string }) => {
    // 1. Exercise Split — get the two new contractIds
    const result = await client.ledger.commands.exerciseAndGetResult<
      { splitAmount: string },
      { contractId1: string; contractId2: string }
    >(
      'my-package:Iou:Iou',
      vars.contractId,
      'Split',
      { splitAmount: vars.splitAmount },
      ['Alice::122059a10c...'],
    );

    // 2. Return whatever you want — accessible via mutation.data
    return result.result;
  },
  invalidateTemplates: ['my-package:Iou:Iou'],
  onSuccess: (result) => {
    console.log('Split into:', result.contractId1, result.contractId2);
  },
});

// Trigger:
mutate({ contractId: 'abc123', splitAmount: '40' });
```
