## 1. Create a client

```ts
import { createNexusClient } from '@nexus-framework/core';

const nexus = createNexusClient({
  ledgerApiUrl: 'http://localhost:7575',
  auth: {
    type: 'sandbox',
    secret: 'secret',      // Canton Sandbox default
    userId: 'alice',
    partyId: 'Alice::122059a10c...',
  },
});
```

## 2. Query active contracts

```ts
const { contracts } = await nexus.ledger.contracts.fetchActiveContracts({
  templateId: 'my-package:Iou:Iou',
  parties: ['Alice::122059a10c...'],
});

contracts.forEach(c => console.log(c.contractId, c.payload));
```

## 3. Create a contract

```ts
const result = await nexus.ledger.commands.createContract(
  'my-package:Iou:Iou',
  { owner: 'Alice::122059a10c...', amount: '100', currency: 'USD' },
  ['Alice::122059a10c...'],
);

console.log(result.transactionId);
```

## 4. Use with React (Next.js)

Wrap your app with `NexusProvider`:

```tsx
// app/layout.tsx
import { NexusProvider } from '@nexus-framework/react';

export default function Layout({ children }) {
  return (
    <NexusProvider config={{
      ledgerApiUrl: 'http://localhost:7575',
      auth: { type: 'sandbox', secret: 'secret', userId: 'alice', partyId: 'Alice::...' },
    }}>
      {children}
    </NexusProvider>
  );
}
```

Then use hooks in client components:

```tsx
'use client';
import { useContracts, useCreateContract } from '@nexus-framework/react';

export function IouList() {
  const { data, isLoading } = useContracts({ templateId: 'my-package:Iou:Iou' });
  const { mutate: create, isPending } = useCreateContract();

  if (isLoading) return <p>Loading…</p>;

  return (
    <>
      {data?.contracts.map(c => <div key={c.contractId}>{c.payload.amount}</div>)}
      <button disabled={isPending} onClick={() =>
        create({
          templateId: 'my-package:Iou:Iou',
          createArguments: { owner: 'Alice::...', amount: '50', currency: 'EUR' },
          actAs: ['Alice::...'],
        })
      }>
        Create Iou
      </button>
    </>
  );
}
```
