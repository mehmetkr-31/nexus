## 1. Create a client

`createNexus` takes a `plugins` array. At minimum you need one auth plugin.

```ts title="lib/nexus.ts"
import { createNexus, sandboxAuth } from '@nexus-framework/core';

export const nexus = createNexus({
  ledgerApiUrl: 'http://localhost:7575',
  plugins: [
    sandboxAuth({
      secret: 'secret',            // Canton Sandbox default
      userId: 'alice',
      partyId: 'Alice::122059a10c...',
    }),
  ],
});
```

## 2. Query active contracts

```ts
const { contracts } = await nexus.ledger.contracts.fetchActiveContracts({
  templateId: 'my-package:Iou:Iou',
});

for (const c of contracts) {
  console.log(c.contractId, c.payload);
}
```

## 3. Create and exercise

```ts
// Create
const { transactionId } = await nexus.ledger.commands.createContract(
  'my-package:Iou:Iou',
  { owner: 'Alice::...', amount: '100', currency: 'USD' },
  ['Alice::...'],
);

// Exercise and get the Daml return value
const result = await nexus.ledger.commands.exerciseAndGetResult<
  Record<string, unknown>,
  { newContractId: string }
>('my-package:Iou:Iou', contractId, 'Transfer', { newOwner: 'Bob::...' }, ['Alice::...']);

console.log(result.result.newContractId);
```

## 4. Use with React

Create a `nexus` client with the React plugins, then use it in components:

```ts title="lib/nexus-client.ts"
"use client";

import { createNexusClient, sandboxAuth, tanstackQueryPlugin } from '@nexus-framework/react';

export const nexus = createNexusClient({
  baseUrl: 'http://localhost:7575',
  plugins: [
    sandboxAuth({ secret: 'secret', userId: 'alice', partyId: 'Alice::...' }),
    tanstackQueryPlugin(),
  ],
});
```

```tsx title="components/IouList.tsx"
"use client";
import { nexus } from '@/lib/nexus-client';

export function IouList() {
  const { data, isLoading } = nexus.useContracts({ templateId: 'my-package:Iou:Iou' });
  const { mutate, isPending } = nexus.useCreateContract();

  if (isLoading) return <p>Loading…</p>;

  return (
    <ul>
      {data?.contracts.map(c => <li key={c.contractId}>{c.payload.amount}</li>)}
    </ul>
  );
}
```

```tsx title="app/layout.tsx"
import { nexus } from '@/lib/nexus-client';

export default function Layout({ children }) {
  return <nexus.NexusProvider>{children}</nexus.NexusProvider>;
}
```
